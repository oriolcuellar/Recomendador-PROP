427 private boolean binarySearch(ArrayList<Item> Items, int l, int r, int ID) {
428
429 if (r >= l) {
430 int mid = l + (r - l) / 2;
431 //mirem se es en el mig
432 if (Items.get(mid).getID() == ID)
433 return true;
434 //si no en es subvector esquerra
435 if (Items.get(mid).getID() > ID)
