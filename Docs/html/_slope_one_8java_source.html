149 float pivote = A.get(izq).getValoration();

156 while( A.get(i).getValoration() >= pivote && i < j) i++;

158 if (i < j) {

160 auxID = A.get(i).getItemID();
161 myPair mp = new myPair(A.get(j).getItemID(),A.get(j).getValoration());
162 A.set(i,mp);

164 A.set(j,mp1);

167 myPair mp = new myPair(A.get(j).getItemID(),A.get(j).getValoration());
168 A.set(izq,mp);
169 myPair piv = new myPair(pivoteID,pivote);
170 A.set(j,piv);
171
172 if(izq < j-1)
173 quicksort(A,izq,j-1);
174 if(j+1 < der)
175 quicksort(A,j+1,der);
